# 开启warp脚本
bash /a_script/warp.sh 

# 关闭
warp-cli disconnect

# 每天6点/19点 关闭warp-cli
0 6 * * * bash /a_script/warp_close.sh
0 19 * * * bash /a_script/warp_close.sh

# 查看cron定时任务状态
systemctl status cron

# 刷新cron定时任务
sudo systemctl restart cron

# cron设置执行时区
sudo timedatectl set-timezone Asia/Shanghai

# 打开定时任务
sudo crontab -e
